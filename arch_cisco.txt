# Frame Relay
>> DCE - Data Communication Equipment
SW#show frame-relay route  # print routes of frame relay protocol
SW#show frame-relay pvc  # print virtual circuit status of listed frame relay interfaces
SW#show frame-relay map  # print info on static and dynamic ip mappings to dlci(s)
SW(config)#frame-relay switching  # define device frame relay switch

SW(config)#int s0/2/0
SW(config-if)#encapsulation frame-relay
SW(config-if)#frame-relay intf-type dce  # can match different physical interface types
SW(config-if)#clockrate 64000
SW(config-if)#frame-relay route 102 interface s0/2/1 201  # forward traffic from route 102 to serial route 201
SW(config-if)#no shut

>> DTE - Data Terminal Equipment
router#show frame-relay map  # print frame relay addressing info
router#show frame-relay pvc  # print pvc stats
router#show frame-relay lmi  # print lmi stats
router# |no debug frame-relay lmi  # enable|disable debug, lmi packets
router#clear frame-relay inverse-arp  # clears dynamically learned frame relay ip mapping

>> DLCI - Data Link Connection Identifier, CIR - Committed Information Rate
router(config-if)#encapsulation frame-relay cisco|ietf  # define frame relay encapsulation type (DTE-to-DTE)
router(config-if)#frame-relay lmi-type cisco|ansi|q933a  # define lmi encapsulation type (DTE-to-DCE)
router(config-if)#no frame-relay inverse-arp  # disable inverse arp; does not allow dynamic learning of pvc circuits
router(config-if)#ip address IP MASK  # assign ip address to local int
router(config-if)#frame-relay map ip IP DLCI broadcast  # map remote ip to relevant dlci
router(config-if)#frame-relay map ip 10.1.1.2 102 broadcast  # allows ip broadcasts to transit link
router(config-if)#no shut

>> DTE - sub-interfaces
router(config)#int INT
router(config-if)#encapsulation TYPE
router(config-if)#int INT-SUB TYPE
router(config-subif)#ip address IP MASK
router(config-subif)#frame-relay interface-dlci DLCI cisco|ietf  # can have different encap type per subint

router(config)#int fa0/1
router(config-if)#no shut
router(config-if)encapsulation frame-relay
router(config-if)int fa0/1.1 point-to-point
router(config-subif)#ip address 140.1.1.1 255.255.255.0
router(config-subif)#frame-relay interface-dlci 501 ietf

router(config-subif)#int fa0/1.2 multipoint
router(config-subif)#ip address 140.1.2.1 255.255.255.0
router(config-subif)#frame-relay interface-dlci 502
router(config-subif)#frame-relay interface-dlci 503
